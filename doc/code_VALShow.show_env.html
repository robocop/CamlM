let show_env env = 
  let rec show_mods = function
    | [] -> ()
    | (m, (_, _)) :: xs -> print_endline (" --- " ^ m); show_mods xs
  in
  print_endline ("Name : " ^ env.this);
  print_endline ("Loaded stuff : ");
  List.map (function (n, c) -> print_endline (" - Name : " ^ n); show_mods c) env.namespace
