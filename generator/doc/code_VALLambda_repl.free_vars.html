let rec free_vars = function
  | EVariable x -> StringSet.singleton x
  | EFunction {def = [PVariable v, e]; env = _} ->
    StringSet.diff (free_vars e) (StringSet.singleton v)
  | EApplication(m, n) 
  | EPair(m, n)
  | ECons(m, n) ->
      StringSet.union (free_vars m) (free_vars n)
  | ESome n -> free_vars n
  | _ -> StringSet.empty
